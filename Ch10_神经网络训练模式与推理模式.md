

# 神经网络训练模式与推理模式

- model.train开启训练模式；model.eval开启推理模式（推理模式下参数不再更新）
    
    ```python
    #训练模式
    
    def train(epoch, train_batch, valid_batch=None, patience=5):
        model.train()
        samples = 0
        best_loss = float('inf')
        epochs_no_improve = 0
    
        for eps in range(epoch):
            # 训练阶段
            for idx, (x, y) in enumerate(train_batch):
                opt.zero_grad()
                loss = criterion(model(x), y)
                loss.backward()
                opt.step()
                samples += x.shape[0]
                if (idx + 1) % 128 == 0 or idx == len(train_batch) - 1:
                    print(f'epoch:{eps} | {samples}/{(mnist.data.size()[0]) * epoch} | '
                          f'{100 * samples / (mnist.data.size()[0] * epoch):.0f}% | Loss:{loss.item():.4f}')
    
            # 验证和早停
            if valid_batch is not None:
                model.eval()
                val_loss = 0.0
                total_samples = 0
                with torch.no_grad():
                    for x_val, y_val in valid_batch:
                        output = model(x_val)
                        val_loss += criterion(output, y_val).item() * x_val.size(0)
                        total_samples += x_val.size(0)
                val_loss /= total_samples  # 平均验证损失
                print(f'Validation Loss after epoch {eps}: {val_loss:.4f}')
    
                if val_loss < best_loss:
                    best_loss = val_loss
                    epochs_no_improve = 0
                    # 可选：保存最佳模型
                    # torch.save(model.state_dict(), 'best_model.pth')
                else:
                    epochs_no_improve += 1
                    if epochs_no_improve >= patience:
                        print(f'Early stopping triggered after {eps + 1} epochs.')
                        break  # 退出训练循环
                model.train()  # 恢复训练模式
    		
    #推理模式
    model.eval()
    with torch.no_grad():
    	y_train_pred = model(X_train)
    	y_test_pred = model(X_test)
    
    #查看某一层批归一化BatchNorm层的统计量
    model.layer1.running_mean
    model.layer1.running_var
    ```

- 训练过程可视化呈现（TensofrBoard）
    
    ```python
    import torch
    import torch.nn as nn
    import torch.optim as optim
    from torch.utils.data import DataLoader, TensorDataset
    from torch.utils.tensorboard import SummaryWriter
    
    # ---------- 1. 定义MLP模型 ----------
    class MLP(nn.Module):
        def __init__(self, input_dim=10, hidden_dim=20, output_dim=1):
            super().__init__()
            self.net = nn.Sequential(
                nn.Linear(input_dim, hidden_dim),
                nn.ReLU(),
                nn.Linear(hidden_dim, output_dim)
            )
    
        def forward(self, x):
            return self.net(x)
    
    # ---------- 2. 生成随机训练数据 ----------
    # 输入特征维度10，训练样本数200
    X = torch.randn(200, 10)
    # 模拟回归任务：真实标签 = 输入各维之和 + 随机噪声
    y = X.sum(dim=1, keepdim=True) + 0.1 * torch.randn(200, 1)
    
    dataset = TensorDataset(X, y)
    dataloader = DataLoader(dataset, batch_size=32, shuffle=True)
    
    # ---------- 3. 初始化模型、损失函数、优化器和TensorBoard ----------
    model = MLP()
    criterion = nn.MSELoss()
    optimizer = optim.Adam(model.parameters(), lr=0.01)
    
    # 创建TensorBoard记录器，日志将保存在 ./runs/exp_mlp 目录下
    writer = SummaryWriter('runs/exp_mlp')
    
    # （可选）将模型计算图写入TensorBoard（需要一次虚拟输入）
    dummy_input = torch.randn(1, 10)
    writer.add_graph(model, dummy_input)
    
    # ---------- 4. 训练循环 ----------
    num_epochs = 50
    for epoch in range(num_epochs):
        model.train()
        total_loss = 0.0
        for batch_X, batch_y in dataloader:
            optimizer.zero_grad()
            outputs = model(batch_X)
            loss = criterion(outputs, batch_y)
            loss.backward()
            optimizer.step()
            total_loss += loss.item() * batch_X.size(0)  # 累加batch loss
    
        avg_loss = total_loss / len(dataset)  # 整个epoch的平均损失
    
        # ----- 使用TensorBoard记录标量 -----
        writer.add_scalar('Loss/train', avg_loss, epoch)
    
        # ----- （可选）记录每一层的权重直方图 -----
        for name, param in model.named_parameters():
            writer.add_histogram(name, param, epoch)
    
        print(f'Epoch [{epoch+1}/{num_epochs}], Loss: {avg_loss:.4f}')
    
    # ---------- 5. 关闭writer ----------
    writer.close()
    print("训练完成。在终端运行以下命令查看TensorBoard：")
    print("tensorboard --logdir=./runs")
    ```
