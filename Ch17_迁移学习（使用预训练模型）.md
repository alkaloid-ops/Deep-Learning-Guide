

# 迁移学习（使用预训练模型）

- 使用预训练模型，并自定义冻结部分参数

```python
import torch
import torch.nn as nn
import torch.nn.functional as F
import torchvision.models as ptm

resnet50 = ptm.resnet50()

#预训练模型
resnet50_ptm = ptm.resnet50(weights=ptm.ResNet50_Weights.DEFAULT)

resnet50_ptm.conv1.parameters #查看预训练模型架构参数

resnet50_ptm.conv1.weight #查看预训练模型权重参数

resnet50_ptm.conv1.weight.requires_grad #检查预训练模型参数是否冻结

#预训练模型参数冻结
for i in resnet50_ptm.parameters():
    i.requires_grad = False
    
resnet50_ptm.conv1.weight.requires_grad #检查参数是否冻结成功（False为参数冻结）

resnet50_ptm.fc.in_features #查询预训练模型全连接层的输入尺寸
resnet50_ptm.fc.out_features #查询预训练模型全连接层的输出尺寸

#迁移学习构建预训练和自定义架构
class ResNet(nn.Module):
    def __init__(self):
        super().__init__()
        self.pretrained = nn.Sequential(
            resnet50_ptm.conv1,
            resnet50_ptm.bn1,
            resnet50_ptm.relu,
            resnet50_ptm.maxpool,
            resnet50_ptm.layer1,
            resnet50_ptm.layer2
        )
        self.custom = nn.Sequential(
            resnet50.layer3,
            resnet50.layer4,
            resnet50.avgpool
        )
        self.fc = nn.Linear(
            resnet50.fc.in_features, 512
        )
        self.out = nn.Linear(
            in_features=512,
            out_features=10
        )

    def forward(self, x):
        x = self.pretrained(x)
        x = self.custom(x)
        x = torch.flatten(x, 1)
        x = self.fc(x)
        x = self.out(x)
        return x
    
   
model = ResNet()
model.pretrained[0] #查看预训练模型的结构
model.pretrained[0].weight.requires_grad #检查参数冻结情况
   
from torchinfo import summary
#查看模型各层输出的尺寸和参数量
summary(model, input_size=(10,3,224,224), device='cuda' if torch.cuda.is_available() else 'cpu')
```
